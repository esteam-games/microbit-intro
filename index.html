input.onGesture(Gesture.Shake, function () {
    let choice = randint(1, 6)
    if (choice == 1) {
        basic.showString("Yes")
    } else if (choice == 2) {
        basic.showString("No")
    } else if (choice == 3) {
        basic.showString("Maybe")
    } else if (choice == 4) {
        basic.showString("Ask again")
    } else if (choice == 5) {
        basic.showString("Definitely")
    } else {
        basic.showString("Unlikely")
    }
})

Game 4: Clap the Loudest

👏

Who can make the most noise?

Press ↓ to learn more

How to play:

  • Press button A to start measuring
  • Clap as loud as you can!
  • The micro:bit shows the loudness level
  • Compete with friends to see who's loudest!

What you'll learn:

  • 🎤 Using the built-in microphone
  • 📊 Measuring sound levels
  • 🏆 Creating competitive challenges

Perfect for parties and group activities!

MakeCode Blocks 🧱

The loudest sound wins!

input.onButtonPressed(Button.A, function () {
    basic.showString("GO!")
    basic.pause(500)
    let maxSound = 0

    for (let i = 0; i < 100; i++) {
        let soundLevel = input.soundLevel()
        if (soundLevel > maxSound) {
            maxSound = soundLevel
        }
        basic.pause(10)
    }

    basic.showNumber(maxSound)
})

basic.forever(function () {
    let tilt = input.acceleration(Dimension.X)

    if (tilt > 200 || tilt < -200) {
        music.playTone(262, 100)

        basic.showIcon(IconNames.Happy)

Game 6: Design Your Board Game

🎨 🎲

Create your own adventure!

Press ↓ to learn more

What you'll get:

  • 📄 2 post its for game tiles
  • 🟦 Tiny squares to create your game piece
  • 🎲 Your micro:bit as the smart dice!
  • ✏️ Your imagination to design the adventure

How to build it:

  • 🎯 Goal: Create a 20-step path from start to finish
  • 🎨 Design 2 tiles: One with a GOOD effect, one with a BAD effect
  • 📝 Examples: "Move ahead 2 spaces!" or "Go back 1 space!"
  • 🟦 Arrange tiles: Mix everyone's tiles to create the 20-step path
  • 🎲 Place micro:bit: Put it in the center as your smart dice
  • ♟️ Add game pieces: Choose your piece to move along the path

How to play:

  • 1️⃣ Press Button A to roll the micro:bit dice (1-6)
  • 2️⃣ Move your game piece that many spaces forward
  • 3️⃣ Follow the instruction on the tile you land on!
  • 4️⃣ Good tiles help you, bad tiles challenge you!
  • 5️⃣ First player to reach step 20 wins! 🏆

Your creative tiles make every game unique and exciting!

What you'll learn:

  • 🎮 Game design and creativity
  • 🔘 Programming button controls
  • 🎲 Combining tech with hands-on crafts
  • 👥 Creating games others can enjoy
  • 💡 Thinking like an inventor and designer

The best part? You can play it over and over with friends!

Press button A to roll a dice (1-6) for your board game!
